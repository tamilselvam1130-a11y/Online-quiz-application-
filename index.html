<script>

  const quizData = [

    {

      question: "What does HTML stand for?",

      choices: [

        "Hyper Text Markup Language",

        "Hyperlinks and Text Markup Language",

        "Home Tool Markup Language",

        "Hyper Tool Multi Language"

      ],

      correct: 0

    },

    {

      question: "Which language is used for styling web pages?",

      choices: ["HTML", "JQuery", "CSS", "XML"],

      correct: 2

    },

    {

      question: "Which is not a JavaScript Framework?",

      choices: ["Python Script", "JQuery", "Django", "NodeJS"],

      correct: 2

    },

    {

      question: "Which company developed JavaScript?",

      choices: ["Netscape", "Bell Labs", "Sun Microsystems", "IBM"],

      correct: 0

    },

    {

      question: "Inside which HTML element do we put the JavaScript?",

      choices: ["<js>", "<script>", "<javascript>", "<scripting>"],

      correct: 1

    }

  ];

 

  const quizContainer = document.getElementById('quiz');

  const submitBtn = document.getElementById('submit-btn');

  const resultDiv = document.getElementById('result');

 

  function loadQuiz() {

    const output = quizData.map((item, index) => {

      const choicesHtml = item.choices.map((choice, i) =>

        `<li>

          <label>

            <input type="radio" name="question${index}" value="${i}" />

            ${choice}

          </label>

        </li>`

      ).join('');

 

      return `

        <div class="question-block">

          <p class="question">${index + 1}. ${item.question}</p>

          <ul class="choices">${choicesHtml}</ul>

        </div>

      `;

    }).join('');

 

    quizContainer.innerHTML = output;

    resultDiv.innerHTML = '';

  }

 

  function showResult() {

    let score = 0;

    quizData.forEach((item, index) => {

      const selected = document.querySelector(`input[name=question${index}]:checked`);

      if (selected && parseInt(selected.value) === item.correct) {

        score++;

      }

    });

 

    resultDiv.innerHTML = `You scored ${score} out of ${quizData.length}`;

    submitBtn.disabled = true;

  }

 

  submitBtn.addEventListener('click', () => {

    const allAnswered = quizData.every((item, index) =>

      document.querySelector(`input[name=question${index}]:checked`)

    );

 

    if (!allAnswered) {

      alert('Please answer all questions before submitting.');

      return;

    }

 

    showResult();

  });

 

  loadQuiz();

</script>
